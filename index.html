    :carousel_horse: @array-like/rotate

    ArrayLike rotation for JavaScript. See docs.

    import {range} from '@iterable-iterator/range';
import {rotateLeft} from '@array-like/rotate';

const a = Array.from(range(5));
rotateLeft(a, 0, 5, 2);
a; // 2 3 4 0 1
